  function drawLines() {
    const svg = document.getElementById('lines');
    const canvas = svg.parentElement;
    const rect = canvas.getBoundingClientRect();
    const c0 = document.getElementById('n0').getBoundingClientRect();
    const cx = c0.left - rect.left + c0.width / 2;
    const cy = c0.top  - rect.top  + c0.height / 2;
    svg.innerHTML = '';
    const nodes  = ['n1','n2','n3','n4','n5','n6'];
    const colors = ['#6366f1','#a855f7','#f59e0b','#22c55e','#22c55e','#6366f1'];
    nodes.forEach((id, i) => {
      const er = document.getElementById(id).getBoundingClientRect();
      const ex = er.left - rect.left + er.width  / 2;
      const ey = er.top  - rect.top  + er.height / 2;
      const l = document.createElementNS('http://www.w3.org/2000/svg', 'line');
      l.setAttribute('x1', cx); l.setAttribute('y1', cy);
      l.setAttribute('x2', ex); l.setAttribute('y2', ey);
      l.setAttribute('stroke', colors[i]);
      l.setAttribute('stroke-width', '1.5');
      l.setAttribute('stroke-opacity', '0.18');
      l.setAttribute('stroke-dasharray', '3 4');
      svg.appendChild(l);
    });
  }
  setTimeout(drawLines, 80);
  window.addEventListener('resize', drawLines);
</script>
